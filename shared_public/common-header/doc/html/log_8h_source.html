405  static inline void wc_vlog_format_advanced(char const * const file,
406  char const * const function,
407  size_t const line,
408  log_level_t const log_level,
409  char const * const format,
410  va_list * const args)
411  {
412  if ( (log_level > off) && (wc_get_log_level() >= log_level) )
413  {
414  char const * format_buffer = format;
415  char buffer[WC_LOG_FORMAT_BUFFER_SIZE] = { 0 };
416  if ( (wc_get_log_level() == debug) && (file != NULL) ) {
417 
418  char const * function_name = function;
419  if (function_name == NULL) {
420  function_name = "<unknown>";
421  }
422 
423  char const *file_shortend = file;
424  size_t const file_length = strlen(file);
425  char file_buffer[WC_LOG_MAX_FILE_LENGTH + 1] = { 0 };
426  if (file_length > WC_LOG_MAX_FILE_LENGTH)
427  {
428  strncpy(file_buffer + (file_length - WC_LOG_MAX_FILE_LENGTH), file, WC_LOG_MAX_FILE_LENGTH);
429  file_shortend = file_buffer;
430  }
431 
432  char const *function_shortend = function_name;
433  size_t const function_length = strlen(function_name);
434  char function_buffer[WC_LOG_MAX_FUNCTION_LENGTH + 1] = { 0 };
435  if (function_length > WC_LOG_MAX_FUNCTION_LENGTH)
436  {
437  strncpy(function_buffer + (function_length - WC_LOG_MAX_FUNCTION_LENGTH), function_name, WC_LOG_MAX_FUNCTION_LENGTH);
438  function_shortend = function_buffer;
439  }
440 
441  int const print_result = snprintf(buffer, WC_LOG_FORMAT_BUFFER_SIZE, "%s @%s, %u: ", file_shortend, function_shortend, line);
442  WC_ASSERT(print_result >= 0);
443 
444  strncat(buffer, format, (WC_LOG_FORMAT_BUFFER_SIZE - 1));
445  buffer[WC_LOG_FORMAT_BUFFER_SIZE - 1] = '\0';
446  format_buffer = buffer;
447  }
448 
449  char out_buffer[WC_LOG_BUFFER_SIZE] = { 0 };
450  int const print_result = vsnprintf(out_buffer, WC_LOG_BUFFER_SIZE, format_buffer, *args);
451  out_buffer[WC_LOG_BUFFER_SIZE - 1] = '\0';
452  WC_ASSERT(print_result >= 0);
453 
454  if(print_result > 0)
455  {
456  if(print_result > WC_LOG_BUFFER_SIZE)
457  {
458  wc_log_output(log_level, "Log message too long, next log line shortened.");
459  }
460  wc_log_output(log_level, out_buffer);
461  }
462  }
463  }
