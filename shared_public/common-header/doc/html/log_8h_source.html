393  static inline void wc_vlog_format_advanced(char const * const file,
394  char const * const function,
395  size_t const line,
396  log_level_t const log_level,
397  char const * const format,
398  va_list * const args)
399  {
400  if ( (log_level > off) && (wc_get_log_level() >= log_level) )
401  {
402  char const * format_buffer = format;
403  char buffer[WC_LOG_FORMAT_BUFFER_SIZE] = { 0 };
404  if ( (wc_get_log_level() == debug) && (file != NULL) ) {
405 
406  char const * function_name = function;
407  if (function_name == NULL) {
408  function_name = "<unknown>";
409  }
410 
411  char const *file_shortend = file;
412  size_t const file_length = strlen(file);
413  char file_buffer[WC_LOG_MAX_FILE_LENGTH + 1] = { 0 };
414  if (file_length > WC_LOG_MAX_FILE_LENGTH)
415  {
416  strncpy(file_buffer + (file_length - WC_LOG_MAX_FILE_LENGTH), file, WC_LOG_MAX_FILE_LENGTH);
417  file_shortend = file_buffer;
418  }
419 
420  char const *function_shortend = function_name;
421  size_t const function_length = strlen(function_name);
422  char function_buffer[WC_LOG_MAX_FUNCTION_LENGTH + 1] = { 0 };
423  if (function_length > WC_LOG_MAX_FUNCTION_LENGTH)
424  {
425  strncpy(function_buffer + (function_length - WC_LOG_MAX_FUNCTION_LENGTH), function_name, WC_LOG_MAX_FUNCTION_LENGTH);
426  function_shortend = function_buffer;
427  }
428 
429  int const print_result = snprintf(buffer, WC_LOG_FORMAT_BUFFER_SIZE, "%s @%s, %u: ", file_shortend, function_shortend, line);
430  WC_ASSERT(print_result >= 0);
431 
432  strncat(buffer, format, (WC_LOG_FORMAT_BUFFER_SIZE - 1));
433  buffer[WC_LOG_FORMAT_BUFFER_SIZE - 1] = '\0';
434  format_buffer = buffer;
435  }
436 
437  char out_buffer[WC_LOG_BUFFER_SIZE] = { 0 };
438  int const print_result = vsnprintf(out_buffer, WC_LOG_BUFFER_SIZE, format_buffer, *args);
439  out_buffer[WC_LOG_BUFFER_SIZE - 1] = '\0';
440  WC_ASSERT(print_result >= 0);
441 
442  if(print_result > 0)
443  {
444  if(print_result > WC_LOG_BUFFER_SIZE)
445  {
446  wc_log_output(log_level, "Log message too long, next log line shortened.");
447  }
448  wc_log_output(log_level, out_buffer);
449  }
450  }
451  }
