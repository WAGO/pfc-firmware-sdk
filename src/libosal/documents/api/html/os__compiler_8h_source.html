00023 /**
00024  * @defgroup __PACKING Structure Packing
00025  *
00026  * The syntax of several protocols will be described by structures. The size of a
00027  * structure and the position of its members depends on the compiler settings and
00028  * the used processor. To provide always the same syntax, the packing instructions
00029  * of the compiler have to be used. For example, the one byte alignment instruction
00030  * forces the alignment of the structure members to one. Normally the alignment of
00031  * the members depend on the bit size of the used processor. For example, a 32bit
00032  * processor has a 4byte alignment, a 64bit processor a 8byte boundary.
00033  *
00034  * Based on these information a structure, which contains two members of type
00035  * unsigned char, has a size of 8 bytes on a 32 bit processor. Both members are
00036  * aligned to the 4byte boundary. On a 64bit processor the structure size is
00037  * 8 bytes. Both members are aligned to the 8byte boundary. To force the
00038  * compiler to use a one byte alignment for the structure, the structure size is
00039  * always 2 bytes and the members are aligned to a one byte boundary.
00040  *
00041  * For each structure, which will be used to exchange information with remote systems,
00042  * the alignment instructions have to be used.
00043  *
00044  * The following example shows a typical definition of a structure uses the
00045  * one byte alignment instructions.
00046  *
00047  * @code
00048  *
00049  * #if (__PACKING_PRAGMA_USE != 0)
00050  * #pragma __PACKING_ALIGN1_PRAGMA_ENTER(enter_example)
00051  * #endif
00052  *
00053  * typedef struct example_syntax_tag      example_syntax_t;
00054  *
00055  * __PACKING_ALIGN1_PRE
00056  * struct __PACKING_ALIGN1_POST example_syntax_tag
00057  * {
00058  *   uint16_t        value16;
00059  *   uint8_t         value8;
00060  *   uint32_t        value32;
00061  * };
00062  *
00063  * #if (__PACKING_PRAGMA_USE != 0)
00064  * #pragma __PACKING_ALIGN1_PRAGMA_LEAVE(enter_example)
00065  * #endif
00066  *
00067  * @endcode
00068  *
00069  * The used directives are described in the subsequently listed
00070  * instructions.
00071  */

00084 /**
00085  * The define specifies the alignment instruction type.
00086  * Several compilers use pragma pack information such like the MICROSOFT compilers.
00087  * In this case the value of this macro has to be set to 1. To enter a one byte
00088  * alignment section, set the @ref __PACKING_ALIGN1_PRAGMA_ENTER macro to a compiler
00089  * specific pragma pack instruction. To leave a one byte alignment section,
00090  * set the @ref __PACKING_ALIGN1_PRAGMA_LEAVE macro to a compiler specific pragma
00091  * pack instruction.
00092  *
00093  * Compilers, which do not support the pragma pack directives, the value has to be
00094  * set to 0. The GNU compilers for example supports attributes. The attribute is
00095  * placed behind the struct directive. The @ref __PACKING_ALIGN1_POST macro has to
00096  * be set to the compiler specific alignment instruction.
00097  *
00098  * The REALVIEW compilers for example use attributes to force the one byte alignment,
00099  * too. But the attribute is placed in front to the struct directive. In this case
00100  * the @ref __PACKING_ALIGN1_PRE macro has to be set to the compiler specific alignment
00101  * instruction. The value of the @ref __PACKING_PRAGMA_USE macro is set to 0.
00102  */
00103 #define __PACKING_PRAGMA_USE                      (0)
00104 
00105 /**
00106  * For compilers, which uses a packing attribute in front of the struct directives,
00107  * the @ref __PACKING_ALIGN1_PRE has to be set to the compiler specific directives.

00109  *
00110  * @code

00112  * @endcode
00113  *
00114  * @note
00115  * For compilers, which do not support the packing attribute in front of struct
00116  * directives, the macro is empty.
00117  */
00118 #define __PACKING_ALIGN1_PRE

00123  * The GNU compilers uses the @e __attribute__((__packed__)) attribute for example:
00124  *
00125  * @code
00126  * #define __PACKING_ALIGN1_POST                     __attribute__((__packed__))
00127  * @endcode
00128  *
00129  * @note
00130  * For compilers, which do not support the packing attribute behind struct
00131  * directives, the macro is empty.
00132  */
00133 #define __PACKING_ALIGN1_POST                     __attribute__((__packed__))

00135 /**
00136  * Use the definition only for compilers supports the pragma pack directives.
00137  * By using MICROSOFT compilers the macro has to set to following directives:
00138  *
00139  * @code
00140  * #define __PACKING_ALIGN1_PRAGMA_ENTER(lable) pack( push, lable, 1)
00141  * @endcode
00142  *
00143  * The value 1 specifies an one byte alignment.
00144  *
00145  * @note
00146  * The current packing alignment will be stored to the compiler stack by using
00147  * the pragma pack directive with the @e push argument. To restore the currently
00148  * used packing alignment the pragma pack directive with the @e pop argument
00149  * (@ref __PACKING_ALIGN1_PRAGMA_LEAVE) has to be used.
00150  *
00151  * @note
00152  * For compilers, which do not support pragma pack directives, the macro is empty.
00153  */
00154 #define __PACKING_ALIGN1_PRAGMA_ENTER(lable)

00163  *
00164  * The value 1 specifies an one byte alignment.

00171  *
00172  * @note
00173  * For compilers, which do not support pragma pack directives, the macro is empty.
00174  */
00175 #define __PACKING_ALIGN1_PRAGMA_LEAVE(lable)
