00073 /**
00074  * @defgroup __TCHAR_ System Independent Character Type
00075  *
00076  * Normally the systems uses characters to define a string. If another languages
00077  * such like Japanese or Chinese have to be supported, the system has to use
00078  * width characters (WCHAR).
00079  *
00080  * For example the MICROSOFT compilers supports the _UNICODE define to
00081  * use width character strings. UNICODE routines such like @e wcslen
00082  * has to be used instead of standard routines such like @e strlen.
00083  *
00084  * To be independent to the standard format or the UNICODE format, the system
00085  * provides TCHAR types and TCHAR routines. For example the MICROSOFT
00086  * standard libraries provides the routine @e _tcslen, which is mapped to
00087  * the @e strlen routine, if the UNICODE define is not set. Otherwise the
00088  * routine is mapped to the UNICODE routine @e wcslen. The type tchar_t
00089  * is mapped to the basic type @e char (standard) or to the basic type
00090  * @e unsigned @e char (UNICODE).
00091  *
00092  * Macros are required to define static strings depend on the global settings
00093  * (UNICODE). Normally a static string will be defined shown in
00094  * the following example:
00095  *
00096  * @code
00097  * static char string_x[] = "example";
00098  * @endcode
00099  *
00100  * The variable @e string_x can not be used for UNICODE operations. To
00101  * be independent on the UNICODE settings, the string can be defined
00102  * shown in the following example:
00103  *
00104  * @code
00105  * static tchar_t string_x[] = _TEXT("example");
00106  * @endcode
00107  *
00108  * The string variable can be used without changes in a standard system
00109  * and an UNICODE system.
00110  *
00111  * @note
00112  * Maybe the macros are already defined by the operating system.
00113  * Please use the macros to define strings have to be provided by
00114  * calling routines of the operating systems. For example, the
00115  * operating system MS Windows CE requires system independent
00116  * strings of type tchar_t* (LPCTSTR).
00117  *
00118  * @{
00119  */
00120 
00121 #ifdef _UNICODE
00122 #define __T(x) L ## x
00123 #else /* ndef _UNICODE */
00124 #define __T(x) x
00125 #endif
00126 
00127 /**
00128  * Specifies a macro maps the @e __T macro to define independent strings.
00129  * The argument x specifies the string.
00130  */
00131 #define _T(x) __T(x)
00132 
00133 /**
00134  * Specifies a macro maps the @e __T macro to define independent strings.
00135  * The argument x specifies the string.
00136  */
00137 #define _TEXT(x) __T(x)
00138 
00139 /**
00140  * Specifies a macro maps the @e __T macro to define independent strings.
00141  * The argument x specifies the string.
00142  */
00143 #define TEXT(x)  __T(x)

00558 /**
00559  * @defgroup os_csection Critical Section References
00560  *
00561  * Critical sections provide synchronization similar to the functionality of mutex
00562  * objects. A section can be used to protect resources shared between separate threads
00563  * from being used simultaneously.
00564  *
00565  * Normally critical section objects provide a slightly faster, more efficient mechanism
00566  * for mutual-exclusion synchronization. But these objects can be used only by the
00567  * threads of a single process. Event, mutex, and semaphore objects can also be used in
00568  * a single-process application. Like a mutex object, a critical section object can be
00569  * owned by only one thread at a time, which makes it useful for protecting a shared
00570  * resource from simultaneous access. Unlike a mutex object, there is no way to tell
00571  * whether a critical section has been abandoned.
00572  *
00573  * A critical section object is a special resource of a subset of operating systems. The
00574  * functionality is not provided by each operating system. In this case the abstracted
00575  * routines uses the functionality of a real mutex object.
00576  *
00577  * Routines:
00578  *  - @ref os_critical_section_create Create critical section
00579  *  - @ref os_critical_section_destroy Destroy critical section
00580  *  - @ref os_critical_section_lock Lock critical section
00581  *  - @ref os_critical_section_unlock Unlock critical section
00582  */
00583 
00584 /**
00585  * @defgroup os_critical_section_create os_critical_section_create
00586  * @ingroup os_csection
00587  * @{
00588  */
00589 
00590 /**
00591  * The @ref os_critical_section_create routine initializes a critical section object, before it
00592  * can be used. A critical section which is no longer required should be cleaned up by
00593  * calling @ref os_critical_section_destroy.
00594  *
00595  * After a critical section object has been initialized, the threads of the process can
00596  * specify the object in the @ref os_critical_section_lock and the @ref os_critical_section_unlock
00597  * routine to provide mutually exclusive access to a shared resource. For similar
00598  * synchronization between the threads of different processes, use a mutex object.
00599  *
00600  * @return
00601  * A handle to the critical section object indicates success, NULL indicates failure.
00602  *
00603  * @sa
00604  * os_critical_section_destroy os_critical_section_lock os_critical_section_unlock
00605  */
00606 handle_t os_critical_section_create (void);
00607 
00608 /**
00609  * @}
00610  * @defgroup os_critical_section_destroy os_critical_section_destroy
00611  * @ingroup os_csection
00612  * @{
00613  */

00616  * Any thread of the process can use the @ref os_critical_section_destroy routine to release the
00617  * system resources were allocated when the critical section object was initialized.
00618  * After this os_critical_section_destroy has been called, the critical section object can no
00619  * longer be used for synchronization.
00620  *
00621  * @param section     A pointer to the critical section object. The object must have
00622  *                    been previously initialized with the @ref os_critical_section_create
00623  *                    routine.
00624  * @return
00625  *
00626  * @sa
00627  * os_critical_section_create
00628  */

00643  *
00644  * For each time a thread locked a critical section, the thread must call

00646  *
00647  * A critical section which is not longer required should be cleaned up by calling
00648  * @ref os_critical_section_destroy.

00651  *                    been previously initialized with the @ref os_critical_section_create
00652  *                    routine.

00654  *

00667 /**
00668  * A call to the @ref os_critical_section_unlock routine releases the ownership of the specified
00669  * critical section object acquired by a call to @ref os_critical_section_lock. For each time a
00670  * thread locked a critical section, the thread must call @ref os_critical_section_unlock. If a

00893 /* OS Semaphore References */
00894 
00895 /**
00896  * @defgroup os_semaphore Semaphore References
00897  *
00898  * Counting semaphores are a synchronization primitive that allow threads to wait until
00899  * an event has occurred. A counter associated with a semaphore keeps track of the number
00900  * of events that have not yet been processed. The state of the semaphore is
00901  * non-signaled, when this counter is zero. A consumer thread waiting on the semaphore
00902  * will block until another thread posts a new event to the semaphore. The semaphore's
00903  * counter will be incremented. The state is set to signaled, if the counter is greater
00904  * than zero. A thread waiting on the semaphore will consume one event. The semaphore's
00905  * counter will be decremented at this moment.
00906  *
00907  * Typically, an application uses a semaphore to limit the number of threads using a
00908  * resource. The counter would correspond to how many of a certain type of resource are
00909  * currently available. A thread waits on the semaphore to claim a resource or posts the
00910  * semaphore to release the resource again.
00911  *
00912  * To use the functionality of semaphores independent on an operating system the
00913  * semaphore specific routines, constants, types and function prototypes of the
00914  * abstraction layer interface has to be use.
00915  *
00916  * Routines:
00917  *  - @ref os_semaphore_create Create semaphore
00918  *  - @ref os_semaphore_destroy Destroy semaphore
00919  *  - @ref os_semaphore_post Post semaphore event
00920  *  - @ref os_semaphore_wait Wait for semaphore event
00921  */
00922 
00923 /**
00924  * @defgroup os_semaphore_create os_semaphore_create
00925  * @ingroup os_semaphore
00926  * @{
00927  */
00928 /**
00929  * Creating a counting semaphore will be performed by a call to the
00930  * @ref os_semaphore_create routine. The initial counter value specified in the
00931  * @e countInit parameter can be any number, zero or positive, but typically a value of
00932  * zero is used to indicate that no events have occurred yet. In this case the semaphore
00933  * is non-signaled.
00934  *
00935  * Each time a waiting thread is released because of the semaphores signaled state, the
00936  * counter of the semaphore is decreased by 1. A next call to @ref os_semaphore_wait
00937  * a thread will be suspended, if the count is zero. A thread can call
00938  * @ref os_semaphore_post to increment the semaphore's count.
00939  *
00940  * The semaphore remains in the system until it will be removed by a call to
00941  * @ref os_semaphore_destroy.
00942  *
00943  * @param name        Specifies the name of the semaphore object. If @e name is NULL, the
00944  *                    semaphore object is created without a name.
00945  * @param count_init  Specifies an initial count for the semaphore. This value must be
00946  *                    greater than or equal to zero and less than or equal to @e countMax.
00947  * @param count_max   Specifies the maximum count for the semaphore object. This value
00948  *                    must be greater than zero.
00949  *
00950  * @return
00951  * A reference to the created semaphore indicates success. NULL indicates failure.
00952  *
00953  * @sa
00954  * os_semaphore_delete os_semaphore_post os_semaphore_wait
00955  */

00961  * @ingroup os_semaphore
00962  * @{
00963  */
00964 
00965 /**
